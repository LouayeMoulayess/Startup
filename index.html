/* ── SCRATCHPAD ── */ const ScratchpadWidget = { dom: {}, timeout: null, init() { // Centralized DOM caching for better performance this.dom = { input: document.getElementById('scratchpad-input'), indicator: document.getElementById('save-indicator'), clearBtn: document.getElementById('clear-btn'), copyBtn: document.getElementById('copy-btn'), pasteBtn: document.getElementById('paste-btn') }; if(!this.dom.input) return; this.dom.input.value = StorageUtils.get('salut_scratchpad') || ''; this.dom.input.addEventListener('input', (e) => { // Auto-capitalize first letter seamlessly const val = this.dom.input.value; if (val.length === 1 && val !== val.toUpperCase() && e.inputType !== 'deleteContentBackward') { this.dom.input.value = val.toUpperCase(); } clearTimeout(this.timeout); if(this.dom.indicator) this.dom.indicator.classList.remove('visible'); this.timeout = setTimeout(() => { StorageUtils.set('salut_scratchpad', this.dom.input.value); // Dynamic Save Indicator let saveText = "Saved (Local)"; if (window.saveScratchpadToCloud && window.isCloudConnected) { window.saveScratchpadToCloud(this.dom.input.value); saveText = "Saved (Cloud)"; } if(this.dom.indicator) { this.dom.indicator.textContent = saveText; this.dom.indicator.classList.add('visible'); setTimeout(() => { if(this.dom.indicator) this.dom.indicator.classList.remove('visible'); }, CONSTANTS.INDICATOR_DELAY_MS); } }, CONSTANTS.SCRATCHPAD_DEBOUNCE_MS); }); if(this.dom.clearBtn) { // Lazy load the cloud connection to keep the initial page paint snappy if (window.requestIdleCallback) { requestIdleCallback(() => { MarketWidget.init(); WeatherWidget.init(); }); } else { window.addEventListener('load', () => { MarketWidget.init(); WeatherWidget.init(); }); }
